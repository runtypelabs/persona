Quick Start

Three lines to a working chat

Install, configure, and ship. That's it.

import { initAgentWidget } from "@runtypelabs/persona";

// Mount an AI chat widget on any element
initAgentWidget({
  target: document.getElementById("chat"),
  config: {
    apiUrl: "https://your-api.com/chat",
    features: { streaming: true },
    theme: { primary: "#5B2EFF" },
  },
});
